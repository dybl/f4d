project(face4d)
cmake_minimum_required(VERSION 3.8.12)
set(face4d_VERSION_MAJOR 0)
set(face4d_VERSION_MINOR 1)
set(face4d_VERSION_PATCH 0)
set(face4d_VERSION ${face4d_VERSION_MAJOR}.${face4d_VERSION_MINOR}.${face4d_VERSION_PATCH})

SET(CMAKE_INCLUDE_CURRENT_DIR ON)
# SET(CMAKE_AUTOMOC ON)

FIND_PACKAGE(Qt5Widgets)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/include
					${CMAKE_CURRENT_SOURCE_DIR}/src
					${CMAKE_CURRENT_BINARY_DIR}
					)
					
AUX_SOURCE_DIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}/src DIR_SRCS)

FILE(GLOB INCS "include/*.h")
FILE(GLOB HEADERS "src/*.h")
SOURCE_GROUP("Header Files" FILES ${INCS} ${HEADERS})

FILE(GLOB MOCS "include/f4main.h")
FILE(GLOB FORMS "ui/*.ui")
# FILE(GLOB RES "qrc/*.qrc")

SOURCE_GROUP("Form Files" FILES ${FORMS})

QT5_WRAP_CPP(HEADERS_MOC ${MOCS})
QT5_WRAP_UI(FORMS_UIC ${FORMS})
# QT5_ADD_RESOURCES(RES_RCC ${RES})

SOURCE_GROUP("Generated Files" FILES ${HEADERS_MOC} ${FORMS_UIC} "face4d_automoc.cpp")

ADD_EXECUTABLE(face4d ${DIR_SRCS} ${HEADERS} ${INCS} ${HEADERS_MOC} ${FORMS_UIC})

TARGET_LINK_LIBRARIES(face4d Qt5::Widgets)

SET_TARGET_PROPERTIES(face4d PROPERTIES FOLDER "App")
